  \begin{align*}
    0 + y &= y & 0 \times y &= 0 \\

  \end{align*}
  Show the following:
  \begin{enumerate}[label=(\alph*)]
  \item $\forall x y z. x \times z + y \times z = (x + y) \times z$
  \item $\forall x y z. x \times (y \times z) = (x \times y) \times z$
  \end{enumerate}
\end{prob}
